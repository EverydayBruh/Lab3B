Table2* getItemByVer2(Table2 *table, KeyType2 key, RelType2 release){
    Table2* newTable = getItems(table, key);
    if(newTable->size == 0){return NULL;}
    KeySpace2* arr = newTable->arr;
    for(i = 0; i < newTable->size; i++){
        if(arr[i].rel == release){
            return
        }
    }
}
